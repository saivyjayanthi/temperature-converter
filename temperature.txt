<body>
    <div id="converter">
        <h2>Temperature Converter</h2>
        <label for="celsius">Celsius</label>
        <input type="number" id="celsius" placeholder="Enter temperature in Celsius" />

        <label for="fahrenheit">Fahrenheit</label>
        <input type="number" id="fahrenheit" placeholder="Enter temperature in Fahrenheit" />

        <button id="convertBtn">Convert</button>
    </div>

    <script>
        const celsiusInput = document.getElementById("celsius");
        const fahrenheitInput = document.getElementById("fahrenheit");
        const convertBtn = document.getElementById("convertBtn");

        convertBtn.addEventListener("click", () => {
            const celsiusValue = parseFloat(celsiusInput.value);
            const fahrenheitValue = parseFloat(fahrenheitInput.value);

            if (!isNaN(celsiusValue)) {
                const convertedToFahrenheit = (celsiusValue * 9 / 5) + 32;
                fahrenheitInput.value = convertedToFahrenheit.toFixed(2);
            }

            if (!isNaN(fahrenheitValue)) {
                const convertedToCelsius = (fahrenheitValue - 32) * 5 / 9;
                celsiusInput.value = convertedToCelsius.toFixed(2);
            }
        });
    </script>
